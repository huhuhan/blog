rem ---------------------------------------------------------------------------
rem Start script for the CATALINA Server
rem ---------------------------------------------------------------------------
rem 声明采用UTF-8编码 不需要查看信息，将以下中文注释echo改为rem
chcp 65001

setlocal

echo ----------第一阶段：假设 CATALINA_HOME 没有定义-------------
echo 设置CURRENT_DIR为当前目录；%cd%为windows的当前目录
set "CURRENT_DIR=%cd%"
echo 如果CATALINA_HOME 不等于空，跳转到【gotHome】
if not "%CATALINA_HOME%" == "" goto gotHome
echo 为空，设置CATALINA_HOME等于CURRENT_DIR，即当前目录
set "CATALINA_HOME=%CURRENT_DIR%"
echo 如果CATALINA_HOME目录下的catalina.bat存在，跳转到【okHome】
if exist "%CATALINA_HOME%\bin\catalina.bat" goto okHome
echo 不存在，切换到上级目录
cd ..
echo 设置CATALINA_HOME为当前目录
set "CATALINA_HOME=%cd%"
echo 切换到CURRENT_DIR目录
cd "%CURRENT_DIR%"
echo ----------第一阶段结束，输出配置信息---------------------------
echo CURRENT_DIR："%CURRENT_DIR%"
echo CATALINA_HOME： "%CATALINA_HOME%"

:gotHome
echo ----------第二阶段：gotHome----------------------
echo 如果CATALINA_HOME目录下的catalina.bat存在，跳转到【okHome】
if exist "%CATALINA_HOME%\bin\catalina.bat" goto okHome
echo The CATALINA_HOME environment variable is not defined correctly
echo This environment variable is needed to run this program
echo 跳转到【end】
goto end

:okHome
echo ----------第三阶段：okHome----------------------
echo 定义EXECUTABLE表示catalina.bat
set "EXECUTABLE=%CATALINA_HOME%\bin\catalina.bat"
echo EXECUTABLE文件存在，跳转到【okExec】
rem Check that target executable exists
if exist "%EXECUTABLE%" goto okExec
echo Cannot find "%EXECUTABLE%"
echo This file is needed to run this program
echo 跳转到【end】
goto end


:okExec
echo ----------第四阶段：okExec----------------------
rem Get remaining unshifted command line arguments and save them in the
echo 定义CMD_LINE_ARGS，命令行参数对象
set CMD_LINE_ARGS=

:setArgs
echo ----------第五阶段：setArgs----------------------
echo 第1个参数是否为空，不为空跳转【doneSetArgs】
if ""%1""=="""" goto doneSetArgs
echo 为空，CMD_LINE_ARG后面拼接参数
set CMD_LINE_ARGS=%CMD_LINE_ARGS% %1
echo 移位命令，即将%1赋值给%0，参数2给参数1，%0就是本身脚本xx.bat
shift
goto setArgs

:doneSetArgs
echo ----------第六阶段：doneSetArgs----------------------
echo 执行EXECUTABLE，即%EXECUTABLE%
echo CMD_LINE_ARGS: "%CMD_LINE_ARGS%"
call "%EXECUTABLE%" start %CMD_LINE_ARGS%

echo ----------结束阶段关闭窗口前先暂停，查看信息，按随机键关闭此窗口----------------------
pause
:end
