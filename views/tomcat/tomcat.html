# Tomcat 总结

# 启动日志乱码

修改\conf\logging.properties的编码方式

# 默认
java.util.logging.ConsoleHandler.encoding = UTF-8
# Windows改为GBK
java.util.logging.ConsoleHandler.encoding = GBK

# 开启HTTPS端口

修改\conf\server.xml配置文件，打开个新的服务端口

<!-- http服务，redirectPort跳转端口，默认web.xml里没有开启-->
<Connector port="8080" protocol="HTTP/1.1"
           connectionTimeout="20000"
           redirectPort="8443" />
<!-- https服务，默认配置里是注释的 -->
<Connector port="8443" protocol="org.apache.coyote.http11.Http11NioProtocol"
           maxThreads="150" SSLEnabled="true">
    <SSLHostConfig>
        <!-- 修改证书路径，并加个证书密码属性certificateKeystorePassword -->
        <Certificate certificateKeystoreFile="conf/tomcat.jks"
                     type="RSA" certificateKeystorePassword="111111"/>
    </SSLHostConfig>
</Connector>

	<security-constraint>

            <web-resource-name>SSL</web-resource-name>
            <url-pattern>/*</url-pattern>
        </web-resource-collection>
        <user-data-constraint>
            <transport-guarantee>CONFIDENTIAL</transport-guarantee>
        </user-data-constraint>
    </security-constraint>

# 调优方式

  • 禁用AJP服务，不确定

    功能是将Tomcat与其他HTTP服务器做个长连接，一般是Apache服务器（现在应该不用了）

    如图，Web客户1直接访问Tomcat服务器上的JSP组件，URL为http://localhost:8080/index.jsp。Web客户2通过HTTP服务器访问Tomcat服务器上的JSP组件。假定HTTP服务器使用的HTTP端口为默认的8090端口，那么Web客户2访问的URL为http://localhost:8090/index.jsp

    <!-- 默认配置里有个连接器，采用的AJP协议 -->
<Connector port="8009" protocol="AJP/1.3" redirectPort="8443" />
    1
    2

  • 开启线程池，根据实际情况配置线程池

  • 修改tomcat运行模式，采用最优的IO性能

  <?xml version="1.0" encoding="UTF-8"?>


<Server port="8005" shutdown="SHUTDOWN">
  <Listener className="org.apache.catalina.startup.VersionLoggerListener" />


  <Listener className="org.apache.catalina.core.AprLifecycleListener" SSLEngine="on" />
  <Listener className="org.apache.catalina.core.JreMemoryLeakPreventionListener" />
  <Listener className="org.apache.catalina.mbeans.GlobalResourcesLifecycleListener" />
  <Listener className="org.apache.catalina.core.ThreadLocalLeakPreventionListener" />


  <GlobalNamingResources>

    <Resource name="UserDatabase" auth="Container"
              type="org.apache.catalina.UserDatabase"
              description="User database that can be updated and saved"
              factory="org.apache.catalina.users.MemoryUserDatabaseFactory"
              pathname="conf/tomcat-users.xml" />
  </GlobalNamingResources>


  <Service name="Catalina">
	<!--
		Tomcat调优：自定义线程池配置
		默认注释，有默认数值
		参数说明：
		maxThreads：最大并发数，默认设置 200，一般建议在 500 ~ 1000，根据硬件设施和业务来判断
		minSpareThreads：Tomcat 初始化时创建的线程数，默认设置 25
		prestartminSpareThreads： 在 Tomcat 初始化的时候就初始化 minSpareThreads 的参数值，如果不等于 true，minSpareThreads 的值就没啥效果了
		maxQueueSize，最大的等待队列数，超过则拒绝请求
	 -->
    <Executor name="tomcatThreadPool" namePrefix="catalina-exec-"
        maxThreads="500" minSpareThreads="50" prestartminSpareThreads="true" maxQueueSize="100"/>

	<!-- Tomcat调优：添加自定义线程池 -->
	<!-- Tomcat调优：修改运行模式，选NIO2（AIO），默认BIO -->
    <Connector executor="tomcatThreadPool"
               port="9527"
			   protocol="org.apache.coyote.http11.Http11Nio2Protocol"
               connectionTimeout="20000"
               redirectPort="8443" />

	<!-- Tomcat调优：禁用AJP连接器，采用Nginx+tocmat架构 -->
    <Connector port="8009" protocol="AJP/1.3" redirectPort="8443" />

    <Engine name="Catalina" defaultHost="localhost">
      <Realm className="org.apache.catalina.realm.LockOutRealm">
        <Realm className="org.apache.catalina.realm.UserDatabaseRealm"
               resourceName="UserDatabase"/>
      </Realm>

      <Host name="localhost"  appBase="webapps"
            unpackWARs="true" autoDeploy="true">
        <Valve className="org.apache.catalina.valves.AccessLogValve" directory="logs"
               prefix="localhost_access_log" suffix=".txt"
               pattern="%h %l %u %t &quot;%r&quot; %s %b" />
      </Host>
    </Engine>
  </Service>


  <!-- 多个服务实例 -->
  <Service name="Catalina_QueenWeb">
    <Connector port="9523" protocol="HTTP/1.1"
               connectionTimeout="20000"
               redirectPort="8443" />

    <!-- Tomcat调优：禁用AJP连接器，采用Nginx+tocmat架构 -->
    <!-- <Connector port="8009" protocol="AJP/1.3" redirectPort="8443" /> -->

    <Engine name="Catalina_QueenWeb" defaultHost="localhost">

      <!-- 用户访问认证 -->
      <Realm className="org.apache.catalina.realm.LockOutRealm">
        <Realm className="org.apache.catalina.realm.UserDatabaseRealm" resourceName="UserDatabase"/>
      </Realm>
	  <!-- 
	  服务配置：
	  appBase：webapp_QueenWeb，自动生成的文件夹，项目发布在该目录下（ps：ROOT目录为默认根目录）
	  unpackWARs：是否自动解压项目部门的war包
	  autoDeploy：是否自动发布
	  -->
      <Host name="localhost"  appBase="webapp_QueenWeb" unpackWARs="true" autoDeploy="true">
        <Valve className="org.apache.catalina.valves.AccessLogValve" directory="logs"
               prefix="localhost_access_log" suffix=".txt"
               pattern="%h %l %u %t &quot;%r&quot; %s %b" />
      </Host>
    </Engine>
  </Service>

</Server>

# 开启管理页面

  • 自带的项目，可查看服务运行状态和管理发布的项目

  • 默认访问无法登录，因为配置没有用户。可参考未授权页面提示添加用户配置，如图

  • 配置文件conf/tomcat-users.xml中添加，重启起效

      <role rolename="manager-gui"/>
  <role rolename="manager-script"/>
  <role rolename="manager-jmx"/>
  <role rolename="manager-status"/>
  <user username="tomcat" password="tomcat" roles="manager-gui,manager-script,manager-jmx,manager-status"/>
