    # 简介

    # 特点

    • 高并发：支持2-4w并发连接（异步非阻塞）
    • 内存消耗小：一个Nginx进程仅15-20MB（Nginx+PHP服务器）
    • 负载均衡、健康检查、热部署
    • 开源软件：支持商业用途
    • 稳定性高：反向代理功能，服务器一般不会宕机
    • 节省带宽：支持GZIP压缩，添加浏览器本地缓存head头
    • 请求重写：Rewrite重写请求URL，做转发
    • 参考：博客文档 (opens new window)

    # 扩展-OpenResty

    • OpenResty® (opens new window) 是一个基于 Nginx 与 Lua 的高性能 Web 平台，其内部集成了大量精良的 Lua 库、第三方模块以及大多数的依赖项。
    • 用于方便地搭建能够处理超高并发、扩展性极高的动态 Web 应用、Web 服务和动态网关。
    • 编写lua脚本，作为扩展模块，引入nginx.conf配置文件中，加大nginx本身的高并发（可达百万）
    • 适合场景：openresty+redis（自带redis的lua脚本）
      • 首页预加载数据、商城秒杀、IP限流白名单机制、灰度升级
    • 参考博客 (opens new window)

    # 跨域配置

    • 跨域情况

      • 网络协议不同，如http协议访问https协议 ;
      • 端口不同，如80端口访问8080端口 ;
      • 域名不同，如www.test1.com访问www.test2.com ;
      • 子域名不同，如abc.test1.com访问def.test1.com ;

    • 请求分类

      • 简单请求，即请求方式是GET、POST、HEAD之一，且不能加自定义头信息

        可以重写浏览器自带的一些参数，如下，这些是CROS里规范的头信息
Accept
Accept-Language
Content-Language
注意这个，只能是三种之一
Content-Type （仅text/plain，multipart/form-data，application/x-www-form-urlencoded之一）
DPR
Downlink
Save-Data
Viewport-Width
Width

      • 复杂请求，非简单请求即复杂请求。

      • 跨域时，若是复杂请求，则会先发送一次OPTIONS请求，带回Access-Control-*等头信息判断可跨域

    • 跨域案例

      # nginx配置
# 定义参数，第一个为nginx内置源变量，第2个为自定义变量
map $http_origin $my_cross_list{
	default 0;
	"~http://cross2.yhtt.com" http://cross2.yhtt.com;
	"~http://cross4.yhtt.com" http://cross4.yhtt.com;
}

server {
	listen 80;
	server_name cross1.yhtt.com;

	location / {
		root   html;
		index  index-cross1.html index.htm;
	}
	# 允许跨域请求的域，* 代表所有，建议填写指定域名，多个逗号拼接，always可选。
	# 默认只针对状态码200~300+的，而400~500+如401报错不加下面这个头信息。需要后端处理或加always
	#add_header 'Access-Control-Allow-Origin' '*' always;
	# 允许跨域请求的域，指定域名：cross3.yhtt.com
	#add_header 'Access-Control-Allow-Origin' 'http://cross3.yhtt.com';
	# 允许跨域请求的域，指定集合
	add_header 'Access-Control-Allow-Origin' $my_cross_list;
	
	# 允许请求的header
	add_header 'Access-Control-Allow-Headers' '*';
	
	# 允许带上cookie请求，与Access-Control-Allow-Origin=*不能同时用
	add_header 'Access-Control-Allow-Credentials' 'true';
	
	# 允许请求的方法，比如 GET,POST,PUT,DELETE,OPTIONS
	add_header 'Access-Control-Allow-Methods' *;

	# 跨域时的预请求处理
	if ($request_method = 'OPTIONS') {
		return 204;
	}
}

server {
	listen 80;
	server_name cross2.yhtt.com cross3.yhtt.com cross4.yhtt.com;

	location / {
		root   html;
		index  index-cross2.html index.htm;
	}
}

    # 缓存配置

    # 缓存路径：相对路径cache，即nginx安装目录下的cache目录，多级目录需要手写创建父级
# 缓存命名：cache_zone
# 最大缓存：200兆
# 缓存时长：inactive=60m，即60分钟，默认10分钟，期间无访问会删除缓存文件，访问自动续签
proxy_cache_path cache levels=2:2 keys_zone=cache_zone:30m max_size=200m inactive=60m ; # use_temp_path=off;

upstream cache_server{
  server localhost:18070; # 默认index首页
}

server {
  listen 18700;
  server_name localhost;
  location / {
	# 设置缓存内存，上面配置中已经定义好的cache_zone
    proxy_cache cache_zone; 
	# 缓存状态为200的请求，缓存时长为5分钟。（304就是响应来自缓存文件）
    proxy_cache_valid 200 5m; 
	# 缓存文件的key为请求的URI
    proxy_cache_key $request_uri; 
	# 添加缓存状态：MISS、HIT（即响应来自缓存文件）、EXPIRED
    add_header Nginx-Cache-Status $upstream_cache_status; 
    proxy_pass http://cache_server; 
  }
}

    测试用本地未认证的证书，可以通过在线工具或JDK的keytool生成，比如

    # 反向代理-负载均衡

    # 轮询，默认，适合配置相当，无状态且短平快的服务使用
upstream jiqun{
    # max_fails、fail_timeout一起用，在20秒内最大失败次数为3，达到这现象就认为停机
    # fail_time 停机时长，默认10s，停机结束恢复运行
    server	192.168.0.137:9527 max_fails=3 fail_timeout=20s fail_time=30s;
    # backup 备份服务器，主服务器停机后，才发送到这边
    server	192.168.0.137:9523 backup;
    # down 已永久停机
    server	192.168.0.137:9521 down;
}
# 权重，适合硬件配置差别比较大的情况
upstream jiqun2{
    # weight 默认1，值与被访问的几率成正比
    server	192.168.0.137:9527 weight=8; # 80%几率被访问
    server	192.168.0.137:9523 weight=3; 
}
# IP哈希值，适合有状态服务，比如session，根据客户端IP分配保证请求发送到同一服务器
upstream jiqun3{
    ip_hash; # 不能与backup同时使用，服务器需要剔除，必须手动标记down
    server	192.168.0.137:9527 weight=2;
    server	192.168.0.137:9523 max_fails=3 fail_timeout=20s;
