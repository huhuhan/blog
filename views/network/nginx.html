      # 简介

      # 特点

      • 高并发：支持2-4w并发连接（异步非阻塞）
      • 内存消耗小：一个Nginx进程仅15-20MB（Nginx+PHP服务器）
      • 负载均衡、健康检查、热部署
      • 开源软件：支持商业用途
      • 稳定性高：反向代理功能，服务器一般不会宕机
      • 节省带宽：支持GZIP压缩，添加浏览器本地缓存head头
      • 请求重写：Rewrite重写请求URL，做转发

      # 扩展-OpenResty

      • OpenResty® (opens new window) 是一个基于 Nginx 与 Lua 的高性能 Web 平台，其内部集成了大量精良的 Lua 库、第三方模块以及大多数的依赖项。
      • 用于方便地搭建能够处理超高并发、扩展性极高的动态 Web 应用、Web 服务和动态网关。
      • 编写lua脚本，作为扩展模块，引入nginx.conf配置文件中，加大nginx本身的高并发（可达百万）
      • 适合场景：openresty+redis（自带redis的lua脚本）
        • 首页预加载数据、商城秒杀、IP限流白名单机制、灰度升级
      • 参考博客 (opens new window)

      # 跨域配置

      • 跨域情况

        • 网络协议不同，如http协议访问https协议 ;
        • 端口不同，如80端口访问8080端口 ;
        • 域名不同，如www.test1.com访问www.test2.com ;
        • 子域名不同，如abc.test1.com访问def.test1.com ;

      • 子域名跨域测试

	"~http://cross4.yhtt.com" http://cross4.yhtt.com;

      # 缓存配置

      # 缓存路径：相对路径cache，即nginx安装目录下的cache目录，多级目录需要手写创建父级
# 缓存命名：cache_zone
# 最大缓存：200兆
# 缓存时长：inactive=60m，即60分钟，默认10分钟，期间无访问会删除缓存文件，访问自动续签
proxy_cache_path cache levels=2:2 keys_zone=cache_zone:30m max_size=200m inactive=60m ; # use_temp_path=off;

upstream cache_server{
  server localhost:18070; # 默认index首页
}

server {
  listen 18700;
  server_name localhost;
  location / {
	# 设置缓存内存，上面配置中已经定义好的cache_zone
    proxy_cache cache_zone; 
	# 缓存状态为200的请求，缓存时长为5分钟。（304就是响应来自缓存文件）
    proxy_cache_valid 200 5m; 
	# 缓存文件的key为请求的URI
    proxy_cache_key $request_uri; 
	# 添加缓存状态：MISS、HIT（即响应来自缓存文件）、EXPIRED
    add_header Nginx-Cache-Status $upstream_cache_status; 
    proxy_pass http://cache_server; 
  }
}

      # HTTPS配置

      测试用本地未认证的证书，可以通过在线工具或JDK的keytool生成，比如

      keytool -genkey -v -alias yh.com -keyalg RSA -keystore yh.com.keystore -validity 3650
      1

      keytool生成的文件属于JKS格式，直接适用Tomcat，但Nginx需要转化为PEM格式。同样通过在线工具 (opens new window)转化。

      转化后：yh.com_ca.crt、yh.com.key、yh.com.cer。打开看看，将有2个密钥的文件放在nginx服务器上。

      server {
    listen       18600;
    server_name  localhost;
    #将http请求自动跳转到https上
    return 301 https://$server_name$request_uri;
}

server {
    #https的默认端口443
    listen       443 ssl ; #http2 default_server;
    server_name  localhost;

    #证书路径
    ssl_certificate      my/https/yh.com_ca.crt; # 证书 或 根证书
    ssl_certificate_key  my/https/yh.com.key; # 私钥

    ssl_session_cache    shared:SSL:1m;
    ssl_session_timeout  5m;
	
	ssl_protocols TLSv1 TLSv1.1 TLSv1.2; # 支持ssl协议版本，默认为后三个，主流版本是[TLSv1.2]

    ssl_ciphers  HIGH:!aNULL:!MD5;
    ssl_prefer_server_ciphers  on;

    location / {
        root   html;
		index  index.html index.htm;
    }
}

      # 反向代理-负载均衡

      # 轮询，默认，适合配置相当，无状态且短平快的服务使用
upstream jiqun{
    # max_fails、fail_timeout一起用，在20秒内最大失败次数为3，达到这现象就认为停机
    # fail_time 停机时长，默认10s，停机结束恢复运行
    server	192.168.0.137:9527 max_fails=3 fail_timeout=20s fail_time=30s;
    # backup 备份服务器，主服务器停机后，才发送到这边
    server	192.168.0.137:9523 backup;
    # down 已永久停机
    server	192.168.0.137:9521 down;
}
# 权重，适合硬件配置差别比较大的情况
upstream jiqun2{
    # weight 默认1，值与被访问的几率成正比
    server	192.168.0.137:9527 weight=8; # 80%几率被访问
    server	192.168.0.137:9523 weight=3; 
}
# IP哈希值，适合有状态服务，比如session，根据客户端IP分配保证请求发送到同一服务器
upstream jiqun3{
    ip_hash; # 不能与backup同时使用，服务器需要剔除，必须手动标记down
    server	192.168.0.137:9527 weight=2;
    server	192.168.0.137:9523 max_fails=3 fail_timeout=20s;
}
# 最小连接，适合请求处理时间长短不一造成服务器过载的情况
upstream jiqun4{
    least_conn; # 请求转发给连接数较少的服务器
    server	192.168.0.137:9527 weight=2;
    server	192.168.0.137:9523 max_fails=3 fail_timeout=20s;
    server	192.168.0.137:9521 backup;
}
# 集群的统一网关
server {
    listen       9600;
    server_name  localhost;

    location / {
        proxy_pass http://jiqun;
        #	proxy_set_header Host $http_host;
        #	proxy_set_header X-Real-IP $remote_addr;
        #	proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        #	proxy_set_header X-Forwarded-Proto $scheme;
    }
}

      # 反向代理-百度

      # 反向代理-百度，地址即本地ip:6001
server {
    listen  6001;
    location / {
        proxy_pass https://www.baidu.com;
    }

# 反向代理-百度，地址栏自动跳转至www.baidu.com
upstream uBaidu {
    server www.baidu.com ;
}
server{
    listen 6002;
    server_name localhost;

    root   html;
    index  index.html index.htm index.php;

    location / {
        proxy_pass  http://uBaidu;

        #Proxy Settings
        proxy_redirect     off;
        proxy_set_header   Host     www.baidu.com;  #           $host;不能使用$host变量
        proxy_set_header   X-Real-IP        $remote_addr;
        proxy_set_header   X-Forwarded-For  $proxy_add_x_forwarded_for;
        proxy_next_upstream error timeout invalid_header http_500 http_502 http_503 http_504;
        proxy_max_temp_file_size 0;
        proxy_connect_timeout      90;
        proxy_send_timeout         90;
        proxy_read_timeout         90;
        proxy_buffer_size          4k;
        proxy_buffers              4 32k;
        proxy_busy_buffers_size    64k;
        proxy_temp_file_write_size 64k;
    }
}

      # 反向代理-微服务

      # 定义模块服务，负债均衡
upstream bpm-server {
    server bpm-api:8271 max_fails=3 fail_timeout=30s;
    # docker版本1.19后支持网桥模式的网络映射，端口不用更新修改
    #server bpm-api:8213 max_fails=3 fail_timeout=30s;
}
server {
    #监听端口
    listen       18010;

    #监听地址
    server_name  localhost;

    #charset koi8-r;

    #access_log  logs/host.access.log  main;

    #请求的url过滤，正则匹配，~为区分大小写，~*为不区分大小写。
    # 主静态项目
    location / {
        #根目录
        root   C:\Users\fw\Desktop\local-jar\bpm-ui\dist;
        #设置默认页
        index index.html;
    }

    # 次静态项目
    #bpm-ui静态页面
    location /bpm-ui {
        #根目录
        alias	C:\Users\fw\Desktop\local-jar\bpm-ui\static\bpm-ui;
        index index.html;
    }

    #bpm-ui静态页面请求
    location /yh {
        # yh替换为fw
        rewrite  ^/yh/(.*)$ /fw/$1 break;
        proxy_pass http://bpm-server;
    }

    #admin-api服务
    location /api/agile {
        # api/agile替换为fw
        rewrite  ^/api/agile/(.*)$ /fw/$1 break;
        proxy_pass http://bpm-server;
    }
    
    location /api {
        # 非这IP不可访问
        if ( $remote_addr !~* "^192\.168\.0\.48") {
        	return 403;
        }
        # 所有替换为fw
        rewrite  ^/(.*)$ /$1 break;
        proxy_pass http://hs-api:8899;
    }
}

      <service>
    <id>nginx</id>
    <name>nginx</name>
    <description>nginx</description>
    <executable>%BASE%/nginx.exe</executable> 
    <stopexecutable>%BASE%/nginx.exe -s quit</stopexecutable>
    <stopparentprocessfirst>true</stopparentprocessfirst>
    <depend></depend>
    <logpath>%BASE%/logs</logpath>
    <log mode="roll-by-time">
        <pattern>yyyyMMdd</pattern>
    </log>

      # 乱码日志

      nginx.conf检查（-t）不通过，提示又是乱码，可能是保存的编码不对。

      不要用记事本编辑保存（编码为UTF-8withBOM），用其他编辑器保存改为UTF-8保存即可，

      # 路径报错

      nginx.conf中的文件路径不支持有空格，比如windows下的"D:/Program Files"！路径需要加双引号！
