# 规范概念

# 命名要求

# 表字段

  • 由字母、下划线组成

  • 表达信息明确，尽量用单词命名（多个单词用下划线分隔），某些项目可用拼英首字母（政府项目）

  • 禁止用关键字，最后可加下划线后缀，如：id_，name_

  • 禁止缩写前缀+单词的组合形式，表达信息不明确

  • 必须加注释

  • 正确使用字段类型

  • 尽量做到三范式标准（3NF）概念描述 (opens new window)

    • 1NF：属性不可再分；（即每个字段信息明确，比如，XX数量、XX单价，设计为2个）
    • 2NF：消除了非主属性对于码的部分函数依赖
    • 3NF: 消除了非主属性对于码的传递函数依赖

    总之，面向对象结构尽量拆表，拆到不可再拆，单一属性或属性组可查到唯一记录）

# 单位简介

# 长度计算

  • 1byte (字节) = 8bit(位)；1K = 1024 btye；

  • 不同编码，汉字、字母、标点，1字符等于不同字节，查看对比 (opens new window)

  • 比如：UTF-8 编码：1汉字字符 = 3 字节，1字母字符 = 1字节

  • 实际存储，由编码决定，存储长度需要合理计算

# char

  • 适用于固定长度的字段，不够则尾部补齐空格

  • 检索时会去除尾部空格

  • 比如：数据字典值的字段、手机号（11）、身份证号（18）

# varchar

  • 适用于可变长度的字段
  • 额外拿1字节记录长度，所以默认长度是2^8-1=255字节，最大65535字节

# 用户表

CREATE TABLE `org_user` (
  `id_` varchar(32) NOT NULL COMMENT '主键',
  `username_` varchar(64) NOT NULL COMMENT '账户',
  `nickname_` varchar(128) NOT NULL COMMENT '名称',
  `password_` varchar(255) NOT NULL COMMENT '密码',
  `email_` varchar(64) DEFAULT NULL COMMENT '邮箱',
  `mobile_` varchar(11) DEFAULT NULL COMMENT '手机号码',
  `address_` varchar(255) DEFAULT NULL COMMENT '地址',
  `photo_` varchar(255) DEFAULT NULL COMMENT '头像图片附件ID',
  `sex_` varchar(8) DEFAULT NULL COMMENT '性别：男，女，未知',
  `from_` varchar(64) DEFAULT NULL COMMENT '来源',
  `status_` char(2) NOT NULL DEFAULT '01' COMMENT '状态，数据字典，系统用户状态',
  `create_time_` datetime DEFAULT NULL COMMENT '创建时间',
  `create_by_` varchar(64) DEFAULT NULL COMMENT '创建人',
  `update_by_` varchar(64) DEFAULT NULL COMMENT '更新人',
  `update_time_` datetime DEFAULT NULL COMMENT '更新时间',
  `delete_flag_` tinyint(2) DEFAULT '0' COMMENT '是否删除，0否1是',
  PRIMARY KEY (`id_`),
  UNIQUE KEY `email` (`email_`) USING BTREE,
  UNIQUE KEY `mobile` (`mobile_`) USING BTREE,
  KEY `username` (`username_`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='用户表';

CREATE TABLE `sys_file` (
  `id_` varchar(64) COLLATE utf8mb4_bin NOT NULL COMMENT '主键',
  `name_` varchar(64) COLLATE utf8mb4_bin DEFAULT NULL COMMENT '文件名',
  `ext_` varchar(10) COLLATE utf8mb4_bin DEFAULT NULL COMMENT '扩展名',
  `uploader_` varchar(8) COLLATE utf8mb4_bin DEFAULT NULL COMMENT '上传器',
  `path_` varchar(255) COLLATE utf8mb4_bin DEFAULT NULL COMMENT '路径',
  `hash_` varchar(255) COLLATE utf8mb4_bin DEFAULT NULL COMMENT '哈希值',
  `create_time_` timestamp NULL DEFAULT NULL COMMENT '创建时间',
  `update_time_` timestamp NULL DEFAULT NULL COMMENT '更新时间',
  `delete_flag_` tinyint(2) DEFAULT '0' COMMENT '逻辑标识',
  PRIMARY KEY (`id_`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_bin COMMENT='系统-文件表';

# 系统树、节点

CREATE TABLE `sys_tree` (
  `id_` varchar(64) CHARACTER SET utf8 NOT NULL COMMENT '主键',
  `key_` varchar(64) CHARACTER SET utf8 DEFAULT NULL COMMENT '别名',
  `name_` varchar(256) CHARACTER SET utf8 DEFAULT NULL COMMENT '名字',
  `desc_` varchar(256) CHARACTER SET utf8 DEFAULT NULL COMMENT '描述',
  `system_` tinyint(4) DEFAULT NULL COMMENT '是否系统内置树',
  PRIMARY KEY (`id_`) USING BTREE,
  UNIQUE KEY `key_unique_` (`key_`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='系统树';
