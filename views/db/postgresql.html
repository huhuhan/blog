    # 启动失败

    windows环境

    • 原因：

      • 由于执行命令超时，导致事务锁住。

    • 状况：

      • 会在目录/data/pg_xlog/下会生成事务文件，删除文件。（删不掉提示权限问题，只能重启系统才可删）
      • 会在目录/data下生成pid后缀文件，删除文件

    • 命令查询

      • 重置事务日志

        pg_resetxlog.exe -f ..\data
        1

      • 查看服务状态

        pg_ctl.exe status -D ..\data

# 获取到pid，杀掉进程
taskkill -pid 3242

    # 查询数据库表

    SELECT
	tt.tablename tableName,	-- 表名
	tt.tableowner,	-- 表所属用户
	CAST ( obj_description ( cc.oid  ) AS VARCHAR ) AS tableComment	-- 表注释
FROM
	pg_class cc
	INNER JOIN pg_tables tt ON tt.tablename = cc.relname 
WHERE
	1 = 1 
	AND tt.schemaname = 'sde' -- 模式
ORDER BY
	tt.tablename

    # 查询表字段

    SELECT
	pa.attname AS columnName, -- 字段名
	pt.typname AS dataType,	-- 类型
	pd.description AS columnComment	-- 注释 
FROM
	pg_class AS pc
	INNER JOIN pg_attribute AS pa ON pa.attrelid = pc.oid
	INNER JOIN pg_type AS pt ON pa.atttypid = pt.oid
	LEFT JOIN pg_description AS pd ON pd.objoid = pa.attrelid 
	AND pd.objsubid = pa.attnum 
WHERE
	pc.relname = 'my_user' 
	AND pa.attnum > 0 	-- 自定义表大于0，系统默认表小于0
ORDER BY
	pc.relname DESC,
	pa.attnum ASC

    # 建表语句

    -- 自增序列
CREATE SEQUENCE "public"."demo_id_seq" 
INCREMENT 1
MINVALUE  1
MAXVALUE 9223372036854775807
START 1
CACHE 1;

SELECT setval('"public"."demo_id_seq"', 1, false);
ALTER SEQUENCE "public"."demo_id_seq" OWNER TO "postgres";


-- 建表语句
CREATE TABLE "public"."demo" (
	"id" int4 NOT NULL DEFAULT nextval( 'demo_id_seq' :: regclass ),
	"aa" CHAR (2) COLLATE "pg_catalog"."default",
	"bb" VARCHAR (32) COLLATE "pg_catalog"."default",
	"cc" float8,
	"dd" TIMESTAMP (6),
	"ee" NUMERIC (15, 2),
	CONSTRAINT "demo_pk" PRIMARY KEY ("id") 
);
ALTER TABLE "public"."demo" OWNER TO "postgres";
COMMENT ON COLUMN "public"."demo"."id" IS '主键，依赖于自增序列';
COMMENT ON COLUMN "public"."demo"."aa" IS '字符类型';
COMMENT ON COLUMN "public"."demo"."bb" IS '字符串类型';
COMMENT ON COLUMN "public"."demo"."cc" IS '浮点数类型，float8即double';
COMMENT ON COLUMN "public"."demo"."dd" IS '时间类型，date类型不支持时分秒';
COMMENT ON COLUMN "public"."demo"."ee" IS '数字类型';
COMMENT ON TABLE "public"."demo" IS 'demo表，建表语句参考';


-- 新增字段
ALTER TABLE "public"."demo" ADD COLUMN "ss" char(8) COLLATE "pg_catalog"."default";
COMMENT ON COLUMN "public"."demo"."ss" IS '新增字段';
-- 新增int
ALTER TABLE "public"."demo" ADD COLUMN "xqzmj" int8;

-- 修改字段
ALTER TABLE "public"."demo" ALTER COLUMN "ss" TYPE varchar(255) COLLATE "pg_catalog"."default";
COMMENT ON COLUMN "public"."demo"."ss" IS '修改字段';

-- 删除字段
ALTER TABLE "public"."demo" DROP COLUMN "ss";

    # 备份还原

    • 官方文档：备份用pg_dump (opens new window)，pg_dumpall (opens new window)，还原用psql (opens new window)

    • 参考博客 (opens new window)

    • Windows备份脚本参考如下

      :: 通用语句，关闭回显
@ECHO OFF
:: 设置本地延迟扩展，保证单行命令中多个命名延迟处理
@setlocal enableextensions
:: 通用语句，定位到脚本所在目录
@cd /d "%~dp0"
SET PGPATH="D:\YH\Program Files\PostgreSQL\9.2\bin\pg_dump"
SET SVPATH=D:\YH\Program Files\PostgreSQL\9.2\backup\
:: 外部参数，第1是%1，依次类推
SET PRJDB=%1
SET DBUSR=postgres
SET DBROLE=postgres
:: 获取日期时间值
FOR /F "TOKENS=1,2,3 DELIMS=/ " %%i IN ('DATE /T') DO SET d=%%i-%%j-%%k
FOR /F "TOKENS=1,2,3 DELIMS=: " %%i IN ('TIME /T') DO SET t=%%i%%j%%k
SET DBDUMP=%SVPATH%%PRJDB%_%d%_%t%.dump
@ECHO OFF
:: 备份命令，参数说明参考文档
%PGPATH% -h localhost -p 5432 -U %DBUSR% --role %DBROLE% -v -f "%DBDUMP%" %PRJDB% 
echo Backup Taken Complete %DBDUMP%
:: 暂停命令，测试常用
pause

    window系统

    定时任务：搜索【任务计划程序】-【创建基本任务】配置信息，脚本参考如下

    # forfiles 系统自带命令
forfiles /p %cd% /s /m "xx.*" /d -7 /c "cmd /c del /f @path"
# /p 指定目录，%cd%是变量
# /s 递归搜索子目录
# /m 所有文件类型，指定类型，直接写 *.jpg
# /d 多少天前的文件，-7即七天，命名会根据文件最后修改时间判断
# /c 运行指定命令，即双引号内的命令
# ”cmd /c del /f @path“，其中/c表示最后关闭CMD窗口，强制删除文件
