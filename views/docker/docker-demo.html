# Dockerfile

# springBoot + openjdk

# 基础镜像，使用alpine操作系统，openjkd使用8-jdk-alpine
FROM openjdk:8-jdk-alpine

# 设置 镜像仓库
RUN echo -e "https://mirrors.aliyun.com/alpine/v3.6/main/\n\
https://mirrors.aliyun.com/alpine/v3.6/community" > /etc/apk/repositories
# 安装字体软件
RUN apk --update add ttf-dejavu && rm -rf /var/cache/apk/*

# 设置时间
RUN echo "Asia/Shanghai" >etc/timezone && \
        ln -sf /usr/share/zoneinfo/Asia/Shanghai /etc/localtime
# 系统编码
ENV LANG zh_CN.UTF-8
ENV LANGUAGE zh_CN:zh
ENV LC_ALL zh_CN.UTF-8

# 添加jar包
ADD target/my-springboot.jar app.jar
# 添加本地字体文件夹, 如宋体
ADD font /usr/share/fonts/chinese

#启动容器时的进程
ENTRYPOINT ["java","-jar","/app.jar"]
# 指定配置，配置文件需挂载
#ENTRYPOINT ["java","-jar","/app.jar","--spring.config.location=/opt/bpm/config/prod-82.yml"]
# 指定配置环境变量
#ENTRYPOINT ["java","-jar","/app.jar","--spring.profiles.active=prod"]
#暴露端口
EXPOSE 8866

# vue + node

FROM node:alpine
# 将当前目录下的所有文件（除了`.dockerignore`排除的路径），都拷贝进入 image 文件的`/app`目录。
COPY . /app
# 指定接下来的工作路径为`/app`

# web + node + nginx

多成构建

  • 每条命令代表一个步骤Step。docker内部作为一层，会缓存。
  • 重复构建，相同命令采用缓存构建，构建速度会很快。
  • 因此将package.json先拷贝安装依赖，只要package.json没变化，就采用缓存构建。
  • 一个FROM生成一个镜像，因此第一个是为<none>的缓存镜像，第二个才是最终镜像

# 缓存package

# node镜像
FROM node:alpine as build-stage

# 指定接下来的工作路径为/app  - 类似于cd命令
WORKDIR /app
# 拷贝前端项目到app目录下
COPY . .

# 设置淘宝npm镜像
RUN npm install -g cnpm --registry=https://registry.npm.taobao.org
# 安装依赖
RUN cnpm install

# 打包 - 目的：丢到nginx下跑
RUN cnpm run build

# ======================== 上：npm打包  下：nginx运行 ========================
# nginx镜像
FROM nginx:alpine as production-stage

# 把主机的nginx.conf文件复制到nginx容器的/etc/nginx文件夹下
COPY ./docker/nginx.conf /etc/nginx/nginx.conf
# 拷贝前端vue项目打包后生成的文件到nginx下运行
COPY --from=build-stage /app/dist /usr/share/nginx/html


# 注：CMD不同于RUN，CMD用于指定在容器启动时所要执行的命令，而RUN用于指定镜像构建时所要执行的命令。
#    RUN指令创建的中间镜像会被缓存，并会在下次构建中使用。如果不想使用这些缓存镜像，可以在构建时指定--no-cache参数，如：docker build --no-cache

# 使用daemon off的方式将nginx运行在前台保证镜像不至于退出
CMD ["nginx", "-g", "daemon off;"]

重复执行采用缓存，快速构建 更新package.json，只用重新构建install层和后续命令（Copy等命令仍采用缓存）

# 无缓存

# node镜像
FROM node:alpine as build-stage

# 指定接下来的工作路径为/app  - 类似于cd命令
WORKDIR /app

#先拷贝安装依赖文件，再拷贝项目
COPY ./package-lock.json .
COPY ./package.json .


# 设置淘宝npm镜像
RUN npm install --registry=https://registry.npm.taobao.org
## 安装依赖
#RUN cnpm install

# 拷贝前端项目到app目录下
COPY . .

# 打包 - 目的：丢到nginx下跑
RUN npm run build

# ======================== 上：npm打包  下：nginx运行 ========================
# nginx镜像
FROM nginx:alpine as production-stage

# 把主机的nginx.conf文件复制到nginx容器的/etc/nginx文件夹下
COPY ./docker/nginx.conf /etc/nginx/nginx.conf
# 拷贝前端vue项目打包后生成的文件到nginx下运行
COPY --from=build-stage /app/dist /usr/share/nginx/html

# 注：CMD不同于RUN，CMD用于指定在容器启动时所要执行的命令，而RUN用于指定镜像构建时所要执行的命令。
#    RUN指令创建的中间镜像会被缓存，并会在下次构建中使用。如果不想使用这些缓存镜像，可以在构建时指定--no-cache参数，如：docker build --no-cache

# 使用daemon off的方式将nginx运行在前台保证镜像不至于退出
CMD ["nginx", "-g", "daemon off;"]

# mysql + wordpress

version: '3'
services:
  mysql:
      image: mysql:5.7
      environment:
       - MYSQL_ROOT_PASSWORD=123456
       - MYSQL_DATABASE=wordpress
  web:
      image: wordpress
      links:
       - mysql
      environment:
       - WORDPRESS_DB_PASSWORD=123456
      ports:
       - "8500:80"
      working_dir: /var/www/html
      volumes:
       - wordpress:/var/www/html

# springcloud

  • 同一docker网络：参考博客 (opens new window)、微服务实例 (opens new window)
  • 同一docker网络版本要求：1.19版本+，问题参考 (opens new window)
version: '3'

# 配置同一网络
networks:
  toilet:
    driver: bridge

services:
  toilet-eureka:
    container_name: toilet-eureka
    restart: always
    image: toilet-eureka
    build:
      context: ./
      dockerfile: Dockerfile-eureka
    environment:
      TZ: Asia/Shanghai
    ports:
    # 内部端口映射不用动
      - 8761:8761
    networks:
      - toilet
  toilet-gateway:
    container_name: toilet-gateway
    restart: always
    image: toilet-gateway
    build:
      context: ./
      dockerfile: Dockerfile-gateway
    environment:
      TZ: Asia/Shanghai
    ports:
      - 9000:9000
    networks:
      - toilet
  toilet-auth:
    container_name: toilet-auth
    restart: always
    image: toilet-auth
    build:
      context: ./
      dockerfile: Dockerfile-auth
    environment:
      TZ: Asia/Shanghai
    ports:
      - 9010:9010
    volumes:
      - D:\YH\tmp\toilet-auth\data:/opt/authority-data
    networks:
      - toilet
    # 添加内置hosts映射，application配置中的地址用映射域名
    extra_hosts:
      - toilet-mysql:192.168.0.31
  toilet-system:
    container_name: toilet-system
    restart: always
    image: toilet-system
    build:
      context: ./
      dockerfile: Dockerfile-system
    environment:
      TZ: Asia/Shanghai
    ports:
      - 9011:9011
    volumes:
      - D:\YH\tmp\toilet-system\data:/var/www/data/upload
    networks:
      - toilet
    extra_hosts:
      - toilet-mysql:192.168.0.31
  toilet-map:
    container_name: toilet-map
    restart: always
    image: toilet-map
    build:
      context: ./
      dockerfile: Dockerfile-map
    environment:
      TZ: Asia/Shanghai
    ports:
      - 9012:9012
    networks:
      - toilet
    extra_hosts:
      - toilet-mysql:192.168.0.31
      
  #前端
  toilet-web:
    container_name: toilet-web
    restart: always
    image: toilet-web
    build:
      context: ./
      dockerfile: Dockerfile-web
    environment:
      TZ: Asia/Shanghai
    ports:
    # 内部端口映射不用动，包括ngix的反向代理比如(http://toilet-auth:9010)
      - 9013:80
    networks:
      - toilet
    # 依赖的服务启动后才启动
    depends_on:
      - toilet-gateway
      - toilet-auth
