# 基础命令

docker image xxx、docker container xxx 部分命令都可省略image、container，建议加上，清楚命令所属集;

docker ID 命令可以只打前几个字符，只有唯一就可执行

# 镜像命令

# 列出本机的所有镜像文件。
docker image ls
docker image ls -all

# 删除镜像文件，前提镜像生成的容器需要先删除，否则报错
docker image rm [imageName]

# 拉取镜像，默认官方镜像仓库可省略
docker image pull [仓库镜像带版本号]
docker image pull hello-world
docker image pull registry.cn-shanghai.aliyuncs.com/huhuhan/iot:1.0

# 制作镜像
docker image build -t [镜像名]:[版本号] .
# 指定别名DockerFile构建，指定构建的根目录
docker image build -t [镜像名]:[版本号] -f [Dockfile的绝对路径] [根目录]
docker image build -t testImg:V99 -f /home/test/Dockerfile-my /home/test

# 镜像标记为远程仓库镜像
docker image tag [imageName] [username]/[repository]:[tag]
# 制作远程仓库镜像
docker image build -t [username]/[repository]:[tag] .
# 推送远程仓库镜像
docker push [仓库]/[命名空间]/[镜像]:[版本号]

# 实例：阿里云镜像仓库，需指定参库名，[仓库]/[命名空间]/[镜像]:[版本号]
docker image tag iot:1.0 registry.cn-shanghai.aliyuncs.com/huhuhan/iot:2.0

# 查看镜像构建信息
docker image  history [imageName]

# 离线保存镜像
docker save -o [文件路径] [镜像]
# 离线加载镜像
docker load -i [文件路径]
# 离线创建镜像
docker import [文件路径] [新镜像]

# 容器命令

# 运行镜像，自动生成一个容器，可生成多个；
docker container run hello-world
# 常用参数例子：iot镜像
# --restart always 保证自动重启
# --name [别名] [镜像] 必须写在运行run的参数的后面
# -d 后台挂起运行
# -p 端口映射[外部:内部]
# -v 容器内部挂载主机目录，linux的当前目录写法$PWD，window的当前目录写法%cd%
# -it 容器shell映射到当前窗口，最后加上 /bin/bash 表示执行第一条命令，打开内部shell
# 实例
docker run --name tomcat -p 8080:8080 -v $PWD/test:/usr/local/tomcat/webapps/test -d tomcat:8.5

# 列出本机正在运行的容器，包括终止运行的容器
docker container ls
docker container ls --all

# 启动、停止、删除、停止删除容器，id（不用写全）或上述定义的别名
docker container stop [ID或NAME]
docker container start [ID或NAME]
docker container rm [ID或NAME]
docker container kill [ID或NAME]

# 查看日志 f参数表示查看实时日志
docker container logs [-f] [ID或NAME]
# 复制
docker container cp [source] [ID或NAME]:[targert]
docker container cp [ID或NAME]:[targert] [source]
# 当前目录下的所有文件
docker container cp /. [ID或NAME]:[targert]

# 网络命令

docker network ls
docker network create [NAME]
docker network rm [ID或NAME]
# 某网络添加某容器
docker network connect [网络ID或NAME] [容器ID或NAME]
# 某网络断连某容器，适用于容器端口占用，断开网络端口
docker network disconnect [--force] [网络ID或NAME] [容器ID或NAME]
# 查看当前网络的信息，如连接的容器
docker network inspect [ID或NAME]

# 其他命令

# 进入容器内部linux环境
docker exec -it iot /bin/bash
# 直接控制容器内部命令，如停止gitlab容器
docker exec -it gitlab gitlab-ctl stop

# 查看运行容器
docker ps [--filter name=xxx]
docker top [ID或NAME]
# 监控容器使用情况
docker stats
docker stats --no-stream --format "table{{.Name}}\t{{.CPUPerc}}\t{{.MemUsage}}"
docker stats $(docker ps --format '{{.Names}}')

# 容器卷标
docker volume ls
docker volume inspect [id、name]

# 批量删除镜像
docker rmi `docker image ls --all | grep  "<none>" | awk '{print $3}'`
# windows
docker rmi $(docker images -f "dangling=true" -q)

# 非root用户无权限执行docker，附加权限
# 添加docker用户组
sudo groupadd docker     
# 将登陆用户加入到docker用户组中，$USER表示用户名
sudo gpasswd -a $USER docker 
# 更新用户组   
newgrp docker

# 集群命令

# manager节点命令
# 初始化
docker swarm init
# 获取加入集群的命令
docker swarm join-token worker
docker swarm join-token manager

# 查看节点
docker node ls
# 降级，manager节点改为worker节点
docker node demote [node]
# 升级，worke节点改为manager节点
docker node promote [node]
# 修改节点参数
docker node update [options] [node]
# 删除节点
docker node rm [-f] [node]

# Docker-Compose

# 常用命令

docker-compose --version

docker-compose -h                           # 查看帮助

docker-compose up                           # 创建并运行所有容器
docker-compose up -d                        # 创建并后台运行所有容器
docker-compose -f docker-compose.yml up -d  # 指定模板
docker-compose down                         # 停止并删除容器、网络、卷、镜像。
docker-compose --volumes                    # 回收资源同时删除自定义的卷标，默认不加参数，自定义卷标保留
# 最后追加配置中的services，指定运行某个服务的命令，不支持down，适合up、logs、stop、rm等
docker-compose up -d hello-server

docker-compose logs -f        # 查看容器输出日志，-f 实时
docker-compose pull           # 拉取依赖镜像
dokcer-compose config         # 检查配置
dokcer-compose config -q      # 检查配置，有问题才有输出

docker-compose restart   # 重启服务
docker-compose start     # 启动服务
docker-compose stop      # 停止服务
docker-compose rm -f [serviceA] 	# 删除指定服务，不在询问

docker-compose --compatibility up	#做了资源限制，指定参数

# 脚本参考

参考博客 (opens new window)

# 版本号，不同版本的命名写法略有区别，若部分命令不起效可选降为3或2
version: '3'

# 定义网络，命名格式[当前目录_default]
# 想修改前缀，在同级目录下创建.env文件，输入COMPOSE_PROJECT_NAME=aaa，即[aaa_default]
networks:
  # 默认网络，即[当前目录_default]，当前配置文件中的服务自动添加到这网络中
  default:
    driver: bridge #host, bridge, none   
  # 自定义后缀，即[当前目录_myNet]，当前配置文件中的服务需要指定networks属性才生效
  myNet:
    driver: bridge 
  # 引用现有的网络，[aaa_myNet]，可通过docker network ls命令查看是否存在
  aaa_myNet:
	external: true      
      
# 自定义卷标，数据存储
volumes:
  data1-1:

services:
 #服务节点，适合多个服务启动
  web:
    # 启动后容器名
    container_name: iot-ui
    # 重启策略: always，容器退出时总是重启容器
    # 重启策略: unless-stopped，容器退出时总是重启容器但是不考虑在Docker守护进程启动时就已经停止了的容器
    restart: always
    # image获取远程仓库镜像来运行，若有build属性，则作为构建的镜像名称
    image: iot-demo
    # 构建镜像指定设定上下文根目录，然后以该目录为准指定Dockerfile
    build:
      #执行构建时根目录
      context: ./
      #指定构建用的Dockerfile
      dockerfile: docker/Dockerfile
    environment:
      TZ: Asia/Shanghai
    ports:
      - 9901:8700
    # 网络模式，即默认的模块，查看docker network ls 可得知
    # 若所有服务都配置了网络模式，这个docker-compose则不会生成新的网络
    # 与networks属性，仅一个起效
    network_mode: bridge #host, bridge, none，"service:[service name]"，"container:[container name/id]"
    networks:
      - myNet
    # 连接其他容器，注意是容器名，适用于Nginx配置中以容器名+端口做服务端
    external_links:
        - web-api
    # 保证启动顺序，注意是容器名，必须是当前docker-compose中的
    depends_on:
        - web-api
    # 添加到IP映射到容器内的/etc/hosts
    extra_hosts:
      - "my-mysql:192.168.0.31"
    # 挂载映射
    volumes:
        # 绝对路径，即主机的路径
        - /var/run/docker.sock:/tmp/docker.sock
        # 卷标相对路径
        - data1-1://tmp/test
    #启动后执行命令，会覆盖Dockerfile里的CMD
    command: consul://consul:8500
    # 资源限制，启动命令加参数--compatibility
    deploy:
     resources:
     	# 最大资源，可通过docker stats查看
        limits:
           cpus: '2'
           memory: 2G
        ## 保留的资料
        reservations:
           cpus: '0.5'
           memory: 200M
