        # 项目结构

        微服务项目机构一般如下

        ├─xx-master----------------------------父项目，公共依赖
│  │
│  ├─xx-common--------------------------公共模块，非服务
│  │  │
│  │  ├─xx-common-c1--------------------公共模块，服务一
│  │  │
│  │  ├─xx-common-c2--------------------公共模块，服务二
│  │
│  ├─xx-eureka--------------------------微服务公共服务，比如注册中心
.........

        父子项目配置原则： 0. 父级用springboot创建，mvnw等文件可删除

        1. 非项目，通过maven创建
        2. 项目，通过springboot创建
        3. 删除多余文件夹、文件，比如没用的test目录
        4. 父级创建.gitignore版本忽略文件，适用于所有子项目，并补充忽略文件
        5. 修改说明文件README.md
        6. 分布式项目，各服务打包方式为jar，引用这些服务时，需本地先install，否则去仓库下载会到导致下载失败（除非远程仓库有相应的jar）

        # 父级pom

        <!-- 模块管理版本 -->
<project xmlns="http://maven.apache.org/POM/4.0.0"
         xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
         xsi:schemaLocation="http://maven.apache.org/POM/4.0.0 http://maven.apache.org/xsd/maven-4.0.0.xsd">
    <modelVersion>4.0.0</modelVersion>
    
    <!-- 父级，根pom的父级，比如为springboot版本 -->
    <parent>
        <groupId>org.springframework.boot</groupId>
        <artifactId>spring-boot-starter-parent</artifactId>
         <!-- 后缀加上-SNAPSHOT即快照版本,其他为正式版本，共2种版本，对应2种仓库 -->
         <!-- 不变版本号的情况下，snapshot快照版本会重复下，适合开发情况；本质是maven-metadata-local文件中的时间戳判断 -->
        <!-- 不变版本号的情况下，release正式版本，本地有了就不下了 -->
        <version>2.1.5.RELEASE</version>
        <!-- 本地父子项目删除此配置，默认去仓库下，会找不到，导致maven clean等命令执行失败 -->
       <!-- <relativePath/> <!-- lookup parent from repository -->
    </parent>
    
    <!-- 建议命名方式可以为XX-master，groupId、artifactId唯一标识，建议都不相同，可以按级追加 -->
    <groupId>com.yh.ff</groupId>
    <artifactId>xx—master</artifactId>
    <version>1.0</version>
    <!-- 可执行的服务打jar，公共模块打pom，服务部署打war -->
    <packaging>pom</packaging> 
    <name>xx—master</name>
    <description>xx project for Spring Cloud</description>
    
    <!--项目的名称，Maven产生的文档用，可写，可以和artifactID相同-->   
    <name>xx-master</name>
    
    <!--子项目管理，填artifactID--> 
    <modules>
        <module>xx-aa</module>
    </modules>
    
    <!--本项目的依赖--> 
    <dependencies>
        <dependency>
            <groupId>org.springframework.cloud</groupId>
            <artifactId>spring-cloud-starter-eureka-server</artifactId>
        </dependency>
    </dependencies>
    <!--任何配置是否被传播到子项目-->  
    <inherited/> 
    
    <!-- 子项目依赖信息只有groupID和artifactID的话，其他信息会去匹配父级dependencyManagement下的信息，如version、scope，有助于版本号管理 -->
    <dependencyManagement>
        <dependencies>
            <dependency>
                <groupId>org.mybatis.generator</groupId>
                <artifactId>mybatis-generator-core</artifactId>
                <version>1.3.5</version>
                <version>${mybatis.generator.core.version}</version>
                <scope>test</scope>
            </dependency>
        </dependencies>
        
        <dependency>
            <groupId>org.projectlombok</groupId>
            <artifactId>lombok</artifactId>
            <version>1.16.14</version>
            <!-- 表示依赖不传递，即本项目作为依赖被引用时，这依赖不会加载，适合插件依赖-->
            <optional>true</optional>
        </dependency>
    </dependencyManagement>
    
    <!-- 定义通用属性，可以在整个pom中使用，标签名为key，一般做全局的依赖版本属性控制 -->
    <properties>
        <project.build.sourceEncoding>UTF-8</project.build.sourceEncoding>
        <spring.version>4.1.6.RELEASE</spring.version>
        <mybatis.generator.core.version>1.3.5</mybatis.generator.core.version>
    </properties>
</project>

        # 子级pom

        <!-- 模块管理版本 -->
<project xmlns="http://maven.apache.org/POM/4.0.0"
         xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
         xsi:schemaLocation="http://maven.apache.org/POM/4.0.0 http://maven.apache.org/xsd/maven-4.0.0.xsd">
    <modelVersion>4.0.0</modelVersion>
    <!-- 填父级标识 -->
    <parent>
        <groupId>com.xx.ff</groupId>
        <artifactId>xx-master</artifactId>
        <version>1.0</version>
    </parent>
    <!-- 自身标识, 非特殊独立服务，默认继承父级groupId，可删除 -->
    <groupId>com.xx.ff.aa</groupId>
    <artifactId>xx-aa</artifactId>
    <name>xx-aa</name>
    <description>xx-aa for Spring Cloud</description>
</project>

    <scope>system</scope>
    
    <!-- 本地包所在目录 scope为system -->

        # build

        <!-- 父级标签</project -->
<build>
    <!-- 包的名字 -->
    <finalName>xxx</finalName>
    <!-- 描述项目中资源的位置 -->
    <resources>
        <!-- 自定义资源1 -->
        <resource>
            <!-- 资源目录 -->
            <directory>src/main/java</directory>
            <!-- 包括哪些文件参与打包 -->
            <includes>
                <include>**/*.xml</include>
            </includes>
            <!-- 排除哪些文件不参与打包 -->
            <excludes>
                <exclude>**/*.txt</exclude>
                <exclude>**/*.doc</exclude>
            </excludes>
        </resource>
    </resources>

    <!-- 设置构建时候的插件 -->
    <plugins>
        <plugin>
            <groupId>org.apache.maven.plugins</groupId>
            <!-- 编译插件插件 -->
            <artifactId>maven-compiler-plugin</artifactId>
            <!-- 资源插件（资源的插件） -->
            <artifactId>maven-resources-plugin</artifactId>
            <!-- war插件(将项目打成war包) -->
            <artifactId>maven-resources-plugin</artifactId>
            <version>xx.x</version>
            <configuration>
                <!-- 源代码编译版本 -->
                <source>1.8</source>
                <!-- 目标平台编译版本 -->
                <target>1.8</target>
                <encoding>UTF-8</encoding>
                <!-- war包名字 -->
                <warName>WebMavenDemo1</warName>
            </configuration>
            <executions>
                <execution>
                    <phase>compile</phase>
                </execution>
            </executions>
        </plugin>
        <!--spring boot maven 插件-->
        <plugin>
            <groupId>org.springframework.boot</groupId>
            <artifactId>spring-boot-maven-plugin</artifactId>
            <configuration>
                <!--可执行脚本，内部开发为#!/bin/bash-->
                <executable>true</executable>
                <!--包含本地jar包，参考本地依赖包配置-->
                <includeSystemScope>true</includeSystemScope>
            </configuration>
        </plugin>
    </plugins>
</build>

        # repositories

        <!-- 配置多个仓库, 若项目引用了自己写的jar，然后jar发布在自己构建的maven仓库 -->
<!-- 依赖下载顺序按配置的仓库去下，下不了换下一个，最后在根据本地setting中的配置仓库（镜像中仓库）去下 -->
<!-- 若都下不了，删除本地仓库的失败文件.lastUpdated后缀文件，估计引用了项目中其他服务且又没在远程仓库，需install本地打包进本地仓库 -->
<repositories>
    <!-- GeoSolutions -->
    <repository>
        <id>geosolutions</id>
        <name>GeoSolutions Repository</name>
        <url>http://maven.geo-solutions.it</url>
        <snapshots>
           <!-- 不重复下快照版本，默认快照版本在不变版本号的情况下重复下载 -->
            <enabled>false</enabled>
        </snapshots>
    </repository>
</repositories>
