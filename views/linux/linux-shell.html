# 简介

  • 参考文档：菜鸟教程 (opens new window)
  • 区分 shell 和 shell script 的概念， shell 一般指脚本程序
  • 脚本后缀一般是.sh，仅供辨认，不影响内容

# 解释器

表示能解释执行Shell Script编写的脚本，系统一般自带程序，且已写入path路径。

常见的如下：

  • Bourne Again Shell 即 /bin/bash，容器系统可能没有自带
  • Bourne Shell 即 /bin/sh，更小，容器系统一般会自带

# 运算符

# 用反引号包括运算命令
#注意，运算符前后必须要有空格！！

# bash解释器不直接支持运算，expr是常用的表达式工具。其他比如awk
a=1
b=2
# 相加
val=`expr ${a} + ${b}`
echo "a + b : ${val}"
#以下简化为单行语句，实际建议分行写，逻辑清晰！
# 相减，相乘（注意转义符），整除，取余
echo "a - b : `expr ${a} - ${b}`"
echo "a * b : `expr ${a} \* ${b}`"
echo "a / b : `expr ${a} / ${b}`"
echo "a % b : `expr ${a} % ${b}`"
# 是否相等，输出结果是1或0
echo "a == b : `expr ${a} == ${b}`"
echo "a != b : `expr ${a} != ${b}`"


# 关系运算符，仅支持数字或纯数字字符串
# -eq	是否相等
# -ne	是否不相等
# -gt	是否大于右边
# -lt	是否大于左边
# -ge	是否大于等于右边
# -le	是否大于等于左边

# 布尔运算符
# !		非
# -o	或，有1true即true
# -a	与，都为true即true

# 逻辑运算符
# &&	即AND
# || 	即ORE

# 例子									# 条件判断，[]，中间注意空格间隔
if [ ${a} -lt 5 -o ${b} -gt 100 ]
then
   echo "${a} 小于 5 或 ${b} 大于 100 : 返回 true"
else
   echo "${a} 小于 5 或 ${b} 大于 100 : 返回 false"
fi


# 字符串运算符
# = 	是否相等
# !=	是否不相等
# -z	是否空字符串，即长度等于0
# -n	是否非空字符串
# $		是否为空字符串
#例子
a=
if [ ${a} ]
then
   echo "${a} : 字符串不为空"
else
   echo "${a} : 字符串为空"
fi


# 文件运算符
# -r	是否可读
# -w	是否可写
# -x	是否可执行
# -s	是否为空，即大小是否等于0
# -e	是否存在，包括目录
#----------------------------------------
# -f	是否普通文件，即非目录、非设备文件
# -d 	是否为目录
# -S	是否socket文件
# -L	是否链接文件
# -b	是否块设备文件
# -c	是否字符设备文件
# -p	是否命名管道文件
# -g	是否设置SGID位（UNIX系统，12位）
# -k	是否设置粘着位（UNIX系统，12位）
#例子
file="./demo.sh"
if [ -x ${file} ]
then
   echo "文件可执行"
else
   echo "文件不可执行"

通过ls -l命令查看文件权限分别是文件类型（第9位）和所有者、所属组、其它的读写执行（第0-8位）

例如MySQL的/var/run/mysqld/mysqld.sock，就是socket文件

srwxrwxrwx 1 mysql mysql 0 Aug 17 08:08 mysqld.sock
1

文件特殊权限：

  • suid：即对可执行文件（程序）有可执行权限，则运行时，就有所有的rwx权限。不能直接判断。
  • sgid：标记后，则用户在目录中创建的文件权限等于目录的所在组，不属于用户
  • 粘着位：标记后，用户对目录有w权限，则目录下自己创建的文件就有rwx权限

# 流程控制

# 条件判断
a=10
b=20
# 第一种：多个判断多个结果
if [ ${a} == ${b} ]				
then
   echo "a 等于 b"
elif [ ${a} -gt ${b} ]
then
   echo "a 大于 b"
else
   echo "a 小于 b"
fi
# 第二种：一个判断两个结果
if test -e ./bash							#test命令来判断
then
    echo '文件已存在!'
else
    echo '文件不存在!'
fi
# 第三种：一个判断仅一个结果，不用加else
if  [ ${a} != ${b} ]
then
   echo "a 等于 b"
fi
# 单行写法
if  [ ${a} != ${b} ]; then echo "a 等于 b"; fi


# 条件for循环，注意/bin/sh解释器不支持
for aa in 1 2 3 4 5
do
    echo "The value is: ${aa}"
done

# 条件while循环
bb=1
while(( ${bb} <= 5 ))
do
    echo ${bb}
    let "bb++"								#let命令执行表达式
done

# 条件until循环，和while相反
cc=0
until [ ! ${cc} -lt 10 ]					#注意，结果为false执行，为true结束循环
do
   echo ${cc}
   cc=`expr ${cc} + 1`
done


# 条件选择
echo '你输入的数字为:'
read dd									 	#读取控制台的输入
case ${dd} in								#支持变量和常量数字
    1)  echo '你选择了 1' 					  #以)符号分类	
    ;; 										 #以;;两个分号结束，类似其他语言的break
    2|3)  echo "你选择了 ${dd}"
    ;;
    *)  echo '你没有输入 1 到 3 之间的数字'	  #以*表示默认匹配
    ;;
esac										 #case的字母反写esac，类似if与fi

# 无限循环
while :
do
    echo -n "输入 1 到 3 之间的数字: "
    read ee
    case ${ee} in
        1|2|3) echo "你输入的数字为 ${ee}!"
        ;;
        *) echo "你输入的数字不是 1 到 3 之间的!"
           #break							#终止循环
           continue							#本次循环结束，后续“本轮结束”不会打印
           echo "本轮结束"
        ;;
    esac
done

# 设置成系统服务

  • 要求是可执行程序，比如赋予可执行权限的脚本

  • 将可执行程序配置为系统服务，方便通过service或者systemctl命令操作

    systemctl是Systemd命令集中的一个命令，是CentOS7之后的新系统命令，兼容service、chkconfig；

    service、chkconfig、systemctl命令的关系和区别 (opens new window)

  • 系统快捷目录：/etc/init.d，指向/etc/rc.d/init.d 的软链接

  • 系统服务的默认PID文件路径：/var/run/xx/xx.pid

    • 比如/var/run/my-eureka/my-eureka.pid

  • 系统服务的默认日志文件路径：/var/log/xx.log，与脚本程序日志无关

主要步骤

  1. 可执行程序复制、或软连接映射ln -s，至系统快捷目录
  2. 命令 service xxxx

# Jar包运行

  • spring boot构建jar包，包含web容器，本身就是个可执行文件

    • 提示cannot execute binary file。即非可执行jar包

  • 没权限的话，授权，即ls命令可以看到的颜色为绿色

    ln -s /var/www/my-eureka.jar /etc/init.d/my-eureka

# Tomcat运行

#!/bin/bash
# description: Tomcat9 Start Stop Restart  
# processname: tomcat9  
# chkconfig: 234 20 80  
# tomcat 解压目录
CATALINA_HOME=/ws/app/tomcat/apache-tomcat-9.0.24

case ${1} in
        start)
                ${CATALINA_HOME}/bin/startup.sh
                ;;
        stop)
                ${CATALINA_HOME}/bin/shutdown.sh
                ;;
        restart)
                ${CATALINA_HOME}/bin/shutdown.sh
                ${CATALINA_HOME}/bin/startup.sh
                ;;
		log)
				tail -f ${CATALINA_HOME}/logs/catalina.out
				;;
        *)
                echo 'please use : tomcat {start | stop | restart | log}'
        ;;
esac
exit 0

# Hosts修改

#!/bin/bash
# author:huhuhan

# 参数1：域名
in_url=${1}
# 参数2：IP
in_ip=${2}

updateHost() {
	# 读取域名目前的IP
	inner_host=$(cat /etc/hosts | grep ${in_url} | awk '{print $1}')
	if [ ${inner_host} = ${in_ip} ]; then
		echo "${inner_host}  ${in_url} ok"
	else
		# 替换
		sed -i "s#${inner_host}#${in_ip}#g" /etc/hosts
		if [ $? = 0 ]; then
			echo "change ${inner_host} to ${in_ip} ok"
		else
			# 新增
			inner_ip_map="${in_ip} ${in_url}"
			echo ${inner_ip_map} >>/etc/hosts
			if [ $? = 0 ]; then
				echo "${inner_ip_map} to hosts success host is $(cat /etc/hosts)"
			fi
		fi
	fi
}

main() {
	updateHost
}
main

# 格式异常

运行报错异常如下

/bin/bash^M: bad interpreter: No such file or directory
1

原因是习惯Windows下编辑好脚本上传至服务器，上面的^M就说明了格式不对！

解决方式

  • Notepad++修改的话，选择【编辑】-【文档格式转换】-【转为Unix（LF）】

  • 命令修改

    vi demo.sh
# 最后行显示有[doc]就是windows下保存的格式
