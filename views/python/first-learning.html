      # 条件循环

      # while 循环，支持continue、break
count = 0
while (count < 9):
   print 'The count is:', count
   count = count + 1
 
print "Good bye!"

# while...else 
count = 0
while count < 5:
   print count, " is  less than 5"
   count = count + 1
else:
   print count, " is not less than 5"

# for 循环 ，支持continue、break
fruits = ['banana', 'apple',  'mango']
for fruit in fruits: 
   print '当前水果 :', fruit

for index in range(len(fruits)):
   print '当前水果 :', fruits[index]

# for...else 
for num in range(10,20):  # 迭代 10 到 20 之间的数字
   for i in range(2,num): # 根据因子迭代
      if num%i == 0:      # 确定第一个因子
         j=num/i          # 计算第二个因子
         print '%d 等于 %d * %d' % (num,i,j)
         break            # 跳出当前循环
   else:                  # 循环的 else 部分
      print num, '是一个质数'
        
# pass区别continue和break，仅占位，无作用
for letter in 'Python':
   if letter == 'h':
      pass
      print '这是 pass 块' # 这行代码可删除
   print '当前字母 :', letter
 
print "Good bye!"

      # 类的定义

      class Parent:        # 定义父类
   parentAttr = 100
   __secretCount = 0  # 私有变量
   # 固定方法，表示构造函数，self是固定参数，不用传参，理解为java的this关键字
   def __init__(self):
      print "调用父类构造函数"
 
   def parentMethod(self):
      print '调用父类方法'
 
   def setAttr(self, attr):
      Parent.parentAttr = attr
 
   def getAttr(self):
      print "父类属性 :", Parent.parentAttr
 
class Child(Parent): # 定义子类
   def __init__(self):
      print "调用子类构造方法"
 
   def childMethod(self):
      print '调用子类方法'
 
c = Child()          # 实例化子类
c.childMethod()      # 调用子类的方法
c.parentMethod()     # 调用父类方法
c.setAttr(200)       # 再次调用父类的方法 - 设置属性值
c.getAttr()          # 再次调用父类的方法 - 获取属性值

      __init__：这个双下划线+关键字的是默认方法，比如构造函数，需要重写；

      __XX：前缀双下划线，表示私有变量、私有方法，仅内部的self·对象可以调用

      _xx：即protected作用域，仅单个文件内调用，不支持被其他模块调用
