# 下载

# 介绍

动态路由多数据库源，包括动态路由和多数据库源的功能。

因此实现动态路由数据库源，可以考虑以下几点：

  • 多个数据库源注入Spring容器
  • 多个数据库源集合、当前使用的数据库源
  • 切面或拦截器方式，动态的切换

通常Spring项目集成mybatis或mybatis—plus等框架，使用的默认数据库源都是BeanId为dataSource。

# 动态数据库源的实现

模拟实现一个可自动化配置的动态数据库源包，类似dynamic-datasource-spring-boot-starter (opens new window)

# 多数据库源

  • Druid的数据库源配置，参考官方文档druid-spring-boot-starter (opens new window)

  • Spring JDBC的数据库配置，参考源码DataSourceAutoConfiguration，默认集成的是HikariCP (opens new window)数据库源

    • 第1个抽象类是处理Datasource的父级接口
    • 重点，第2个抽象类AbstractRoutingDataSource
    【AbstractRoutingDataSource源码分析】部分代码，详细源码自己看代码 
    
package org.springframework.jdbc.datasource.lookup;

// 省略 import

public abstract class AbstractRoutingDataSource extends AbstractDataSource implements InitializingBean {
    /** 数据库源，配置对象集合 */
    @Nullable
    private Map<Object, Object> targetDataSources;
    /** 当前数据库源，配置对象 */
    @Nullable
    private Object defaultTargetDataSource;
    private boolean lenientFallback = true;
    private DataSourceLookup dataSourceLookup = new JndiDataSourceLookup();
    /** 数据库源，对象集合 */
    @Nullable
    private Map<Object, DataSource> resolvedDataSources;
    /** 当前数据库源，对象*/
    @Nullable
    private DataSource resolvedDefaultDataSource;



    public void setDataSourceLookup(@Nullable DataSourceLookup dataSourceLookup) {
        this.dataSourceLookup = (DataSourceLookup)(dataSourceLookup != null ? dataSourceLookup : new JndiDataSourceLookup());
    }

    // InitializingBean接口的方法，Spring注入容器时加载的方法
    @Override
    public void afterPropertiesSet() {
		// 主要将targetDataSources、defaultTargetDataSource转化为resolvedDefaultDataSource、resolvedDataSources
    }

    // DataSource接口的方法，获取连接对象
    @Override
    public Connection getConnection() throws SQLException {
        return this.determineTargetDataSource().getConnection();
    }

    // DataSource接口的方法，获取连接对象
    @Override
    public Connection getConnection(String username, String password) throws SQLException {
        return this.determineTargetDataSource().getConnection(username, password);
    }

    // 内部方法，几个接口重写的方法，都会调这个，就是获取最终要用的数据库源对象
    protected DataSource determineTargetDataSource() {
        Assert.notNull(this.resolvedDataSources, "DataSource router not initialized");
        Object lookupKey = this.determineCurrentLookupKey();
        DataSource dataSource = (DataSource)this.resolvedDataSources.get(lookupKey);
        if (dataSource == null && (this.lenientFallback || lookupKey == null)) {
            dataSource = this.resolvedDefaultDataSource;
        }

        if (dataSource == null) {
            throw new IllegalStateException("Cannot determine target DataSource for lookup key [" + lookupKey + "]");
        } else {
            return dataSource;
        }
    }
    
    // 抽象方法，给实现类重写
    @Nullable
    protected abstract Object determineCurrentLookupKey();
}

# 密码加密

  1. 通过jar包生成加密后的密码，命令如下，私钥不用记，重复生成也不一样

    java -cp druid-1.1.10.jar com.alibaba.druid.filter.config.ConfigTools yh123456
    1

  2. spring-boot-starter相关配置如下

    public-key: [公钥]
spring:
  datasource:
    druid:
      url: jdbc:postgresql://xxx.xx.x.xx:5432/test_xx
      driver-class-name: org.postgresql.Driver
      username: yh
      password: [加密后的密码]
      connection-properties: config.decrypt=true;config.decrypt.key=${public-key};druid.stat.mergeSql=true;druid.stat.slowSqlMillis=5000
      # 开启config插件，用于解密，以下两种方式都可以
      filters: config
      filter:
        config:
          enabled: true
