# HttpBasic

  • oauth2认证授权的关键参数client_id
  • 默认是要求通过HttpBasic方式传参的，如下，
  • 以上测试例子，项目配置了security.allowFormAuthenticationForClients(），支持作为parameter传参
  • 可能和后续的token参数Authorization，注意区分，避免覆盖
  • 注意：HttpBasic通过base64加密，是可逆，不过比直接作为parameter暴露要好点
var axios = require('axios');
var qs = require('qs');
var data = qs.stringify({
 'grant_type': 'client_credentials',
'scope': 'all' 
});
// 请求参数错误，浏览器会自动弹窗HttpBasic认证，免得误以为是后端的问题- -！
var config = {
  method: 'post',
  url: 'http://127.0.0.1:9001/oauth/token',
  headers: { 
    //'Authorization': 'Basic ' + window.btoa(client_id + ":" + client_secret)
    'Authorization': 'Basic Y2xpZW50X2RlbW86c2VjcmV0X2NsaWVudF9kZW1v', 
    'Content-Type': 'application/x-www-form-urlencoded', 
    'Cookie': 'JSESSIONID=10879E8D734CB378DD345D654EE9C03B'
  },
  data : data
};

axios(config)
.then(function (response) {
  console.log(JSON.stringify(response.data));
})
.catch(function (error) {
  console.log(error);
});
